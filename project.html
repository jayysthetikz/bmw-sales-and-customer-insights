# Average Price By Transmission
# Price with respect to Mileae and Fuel-Type
# Price Distribution ay boxplot
# Price Distribution with kde

figure , axes = plt.subplots(ncols=2, nrows=2, figsize = (16, 12))

sns.barplot(data=df1, x = 'transmission', y = 'price', ax = axes[0,0], color = 'purple', alpha = 0.2)
axes[0,0].set_title('Transmission V/S Price Plot ~')

sns.scatterplot(data=df1, x = 'price', y = 'mileage', hue = 'fuelType', ax = axes[0, 1], palette='pastel', alpha = 0.5)
axes[0,1].set_title('Price wrt Mileage and Fuel-Type ~')

sns.histplot(data=df1, x = 'price', ax = axes[1,0], bins = 30, color = 'purple', alpha = 0.5, kde=True)
axes[1,0].set_title('Distribution of Prices ~')

sns.boxplot(data=df1, x = 'price', ax = axes[1,1], palette='pastel')
axes[1,1].set_title('Box Plot for Price Distribution ~')

plt.show()
